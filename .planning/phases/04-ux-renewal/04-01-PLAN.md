    In `backend/src/api/v1/schemas/subscription.py` add:

    ```
    class BatchRenewRequest(BaseModel):
        subscription_ids: list[int]

    class BatchRenewSkipped(BaseModel):
        id: int
        reason: str  # "not_found" | "not_active" | "missing_billing_cycle"

    class BatchRenewResponse(BaseModel):
        renewed: list[SubscriptionResponse]
        skipped: list[BatchRenewSkipped]
    ```

    In `backend/src/api/v1/routers/subscriptions.py` import `BatchRenewSubscriptionsUseCase` and the new schemas. Add endpoint BEFORE `@router.get("/{id}")` to avoid path conflicts:

    ```
    @router.post("/batch-renew", response_model=ApiResponse[BatchRenewResponse])
    async def batch_renew(
        body: BatchRenewRequest,
        current_user: User = Depends(require_can_update),
        db: AsyncSession = Depends(get_db),
    ) -> ApiResponse[BatchRenewResponse]:
        repo = SqlSubscriptionRepository(db)
        audit_repo = SqlAuditLogRepository(db)
        uc = BatchRenewSubscriptionsUseCase(repo, audit_repo=audit_repo, actor_user_id=current_user.id, actor_email=current_user.email)
        result = await uc.execute(subscription_ids=body.subscription_ids)
        return ApiResponse.ok(data=BatchRenewResponse(
            renewed=[SubscriptionResponse(**vars(s)) for s in result["renewed"]],
            skipped=[BatchRenewSkipped(**s) for s in result["skipped"]],
        ))
    ```

    Use `require_can_update` per D and project guard convention.
